    cache_key = (folder_path, archive_name)

    # 캐시 히트 조건: 폴더 수정 시간 동일
    if cache_key in _archive_cache:
        cached_mtime, cached_ids = _archive_cache[cache_key]
        try:
            current_mtime = os.path.getmtime(folder_path)
            if current_mtime == cached_mtime:
                # 캐시 사용
                archive_file = os.path.join(folder_path, archive_name)
                with open(archive_file, 'w', encoding='utf-8') as f:
                    for video_id in sorted(cached_ids):
                        f.write(f"{video_id}\n")
                return
